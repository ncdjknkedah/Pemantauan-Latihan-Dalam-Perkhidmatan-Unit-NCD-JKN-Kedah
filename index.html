Langkah Integrasi Google Sheet & Apps Script

Panduan memasang Google Apps Script pada Google Sheet anda

1

Buka Google Sheet Laporan

Buka Google Sheet anda. Klik menu Extensions → Apps Script.

2

Tampalkan Kod Script

Padamkan sebarang kod lalai, kemudian tampalkan kod di bawah sepenuhnya ke dalam editor.

3

Deploy sebagai Web App

Klik Deploy → New Deployment. Pilih jenis Web App. Tetapkan "Execute as: Me" dan "Who has access: Anyone".

4

Salin URL & Tampal di Sini

Selepas deploy, salin Web App URL yang dipaparkan. Tampalkan URL tersebut ke kotak di atas, kemudian klik Simpan URL.

5

Klik "Segarkan Data"

Kembali ke halaman Utama, klik "Segarkan Data" untuk muat turun data terkini dari Google Sheet anda.

Google Apps Script — Code.gs 
// ==========================================
// GOOGLE APPS SCRIPT: INTEGRASI KURSUS NCD 2026
// Kemaskini: 2026
// PENTING: Deploy sebagai Web App dengan akses "Anyone"
// ==========================================

function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheets = ss.getSheets();
  var response = { success: true, staffList: [], data: {} };
  
  for (var i = 0; i < sheets.length; i++) {
    var sheet = sheets[i];
    var name = sheet.getName();
    
    // Abaikan sheet konfigurasi/template
    if (name.toUpperCase() === "CONFIG" || name.toUpperCase() === "TEMPLATE") continue;
    
    response.staffList.push(name);
    
    // Ambil nilai dari sel D3 (Jumlah Jam Terkumpul)
    var totalHours = sheet.getRange("D3").getValue() || 0;
    
    // Ambil rekod dari B5:I
    var lastRow = sheet.getLastRow();
    var trainingList = [];
    
    if (lastRow >= 5) {
      var range = sheet.getRange("B5:I" + lastRow);
      var values = range.getValues();
      
      for (var j = 0; j < values.length; j++) {
        var row = values[j];
        // Hanya rekod yang ada tajuk latihan (Kolum E = indeks 3)
        if (row[3]) {
          var rawDate = row[2];
          var formattedDate = "";
          if (rawDate instanceof Date) {
            formattedDate = Utilities.formatDate(rawDate, Session.getScriptTimeZone(), "yyyy-MM-dd");
          } else {
            formattedDate = rawDate ? rawDate.toString() : "";
          }
          trainingList.push({
            bil:     row[0] || (j + 1),
            jenis:   row[1] || "",
            tarikh:  formattedDate,
            tajuk:   row[3] || "",
            kategori: row[4] || "",
            anjuran: row[5] || "",
            tempat:  row[6] || "",
            tempoh:  parseFloat(row[7]) || 0
          });
        }
      }
    }
    
    response.data[name] = {
      totalHours: parseFloat(totalHours) || 0,
      training: trainingList
    };
  }
  
  return ContentService
    .createTextOutput(JSON.stringify(response))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var params = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName(params.staffName);
    
    if (!sheet) {
      return ContentService
        .createTextOutput(JSON.stringify({ success: false, message: "Tiada sheet bernama '" + params.staffName + "'" }))
        .setMimeType(ContentService.MimeType.JSON);
    }
    
    var lastRow = sheet.getLastRow();
    var nextRow = lastRow < 5 ? 5 : lastRow + 1;
    var bil = nextRow - 4;
    
    sheet.getRange(nextRow, 2, 1, 8).setValues([[
      bil,
      params.jenis   || "",
      params.tarikh  ? new Date(params.tarikh) : new Date(),
      params.tajuk   || "",
      params.kategori || "",
      params.anjuran || "",
      params.tempat  || "",
      parseFloat(params.tempoh) || 0
    ]]);
    
    return ContentService
      .createTextOutput(JSON.stringify({ success: true, message: "Rekod berjaya ditambah!", row: nextRow }))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ success: false, message: "Ralat: " + err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
