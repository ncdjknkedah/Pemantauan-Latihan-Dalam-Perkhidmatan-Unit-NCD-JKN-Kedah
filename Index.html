Sambungan Google Apps Script Web App

Sambungkan sistem secara terus kepada Google Sheet fizikal anda

*Sila buat rujukan kepada panduan di bawah untuk membina Web App API menggunakan Google Apps Script secara percuma.

Langkah Integrasi Google Sheet & Apps Script

Ikuti 5 langkah mudah ini untuk mengaktifkan mod automatik live.

1

Buka Google Sheet Laporan

Buka fail Google Sheet sasaran anda. Klik menu atas Extensions > Apps Script.

2

Tampalkan Kod Script

Padamkan sebarang kod lalai di dalam editor, dan tampalkan kod Google Apps Script yang disediakan di dalam kotak kelabu di bawah.

3

Deploy sebagai Web App

Klik butang Deploy > New Deployment. Pilih jenis Web App.

  • Tetapkan "Execute as" kepada Me (Emel anda).
  • Tetapkan "Who has access" kepada Anyone.
4

Sahkan Kebenaran & Salin URL

Klik "Deploy", benarkan kebenaran akaun anda (*Authorize access*), kemudian salin Web App URL yang dipaparkan.

5

Tampal & Jalankan

Tampalkan URL yang disalin ke dalam kotak input "Web App URL" di bahagian atas halaman ini, kemudian klik "Sambung & Simpan".

Google Apps Script (Code.gs) 
// ==========================================
// GOOGLE APPS SCRIPT: INTEGRASI KURSUS NCD 2026
// Tarikh Kemaskini: 2026-05-22
// ==========================================

function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheets = ss.getSheets();
  var response = {
    success: true,
    staffList: [],
    data: {}
  };
  
  for (var i = 0; i < sheets.length; i++) {
    var sheet = sheets[i];
    var name = sheet.getName();
    
    // Abaikan jika ada sheet khusus rujukan/konfigurasi (contoh: "CONFIG")
    if (name.toUpperCase() === "CONFIG" || name.toUpperCase() === "TEMPLATE") continue;
    
    response.staffList.push(name);
    
    // Ambil nilai sel D3 (Jumlah Jam Kursus Terkumpul)
    var totalHours = sheet.getRange("D3").getValue() || 0;
    
    // Ambil senarai rekod dari B5:I
    var lastRow = sheet.getLastRow();
    var trainingList = [];
    
    if (lastRow >= 5) {
      var range = sheet.getRange("B5:I" + lastRow);
      var values = range.getValues();
      
      for (var j = 0; j < values.length; j++) {
        var row = values[j];
        // Pastikan ruangan tajuk latihan (Indeks ke-3 iaitu Kolum E) mempunyai nilai
        if (row[3]) {
          var rawDate = row[2]; // Kolum D
          var formattedDate = "";
          if (rawDate instanceof Date) {
            formattedDate = Utilities.formatDate(rawDate, Session.getScriptTimeZone(), "yyyy-MM-dd");
          } else {
            formattedDate = rawDate ? rawDate.toString() : "";
          }
          
          trainingList.push({
            bil: row[0] || (j + 1), // Kolum B
            jenis: row[1] || "",    // Kolum C
            tarikh: formattedDate,
            tajuk: row[3] || "",    // Kolum E
            kategori: row[4] || "", // Kolum F
            anjuran: row[5] || "",  // Kolum G
            tempat: row[6] || "",   // Kolum H
            tempoh: parseFloat(row[7]) || 0 // Kolum I
          });
        }
      }
    }
    
    response.data[name] = {
      totalHours: parseFloat(totalHours) || 0,
      training: trainingList
    };
  }
  
  return ContentService.createTextOutput(JSON.stringify(response))
    .setMimeType(ContentService.MimeType.JSON)
    .setHeader("Access-Control-Allow-Origin", "*");
}

function doPost(e) {
  try {
    var params = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName(params.staffName);
    
    if (!sheet) {
      return ContentService.createTextOutput(JSON.stringify({ success: false, message: "Ralat: Tiada nama staf bernama '" + params.staffName + "' di Google Sheet." }))
        .setMimeType(ContentService.MimeType.JSON)
        .setHeader("Access-Control-Allow-Origin", "*");
    }
    
    var lastRow = sheet.getLastRow();
    var nextRow = lastRow < 5 ? 5 : lastRow + 1;
    var bil = nextRow - 4; // Pengiraan Bilangan Automatik
    
    // Tulis data ke range B5:I
    sheet.getRange(nextRow, 2).setValue(bil); // B: Bil
    sheet.getRange(nextRow, 3).setValue(params.jenis); // C: Jenis Latihan
    sheet.getRange(nextRow, 4).setValue(params.tarikh); // D: Tarikh
    sheet.getRange(nextRow, 5).setValue(params.tajuk); // E: Tajuk Latihan
    sheet.getRange(nextRow, 6).setValue(params.kategori); // F: Kategori
    sheet.getRange(nextRow, 7).setValue(params.anjuran); // G: Anjuran
    sheet.getRange(nextRow, 8).setValue(params.tempat); // H: Tempat
    sheet.getRange(nextRow, 9).setValue(parseFloat(params.tempoh)); // I: Tempoh (Jam)
    
    return ContentService.createTextOutput(JSON.stringify({ success: true, message: "Pendaftaran latihan berjaya dikemas kini!" }))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeader("Access-Control-Allow-Origin", "*");
      
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ success: false, message: "Ralat: " + error.toString() }))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeader("Access-Control-Allow-Origin", "*");
  }
}
